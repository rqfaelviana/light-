Trechos de código essenciais

// Explique aqui esse trecho do código
#include <FastLED.h>
#define NUM_LEDS 8
#define DATA_PIN 6
CRGB leds[NUM_LEDS];

// Explique aqui esse trecho do código
void setup() {
  FastLED.addLeds<WS2812, DATA_PIN, GRB>(leds, NUM_LEDS);
}

// Explique aqui esse trecho do código
void loop() {
  for (int i=0; i<NUM_LEDS; i++) { leds[i] = CRGB::Yellow; FastLED.show(); delay(120); leds[i] = CRGB::Black; }
}

Neste bloco, configuramos o temporizador e os padrões para que os LEDs pisquem no mesmo BPM da música. [EDITAR]
